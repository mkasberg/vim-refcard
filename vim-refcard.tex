\section{Starting Vim}
\begin{tabular}{@{}ll@{}}
\verb|vi| \textit{filename} & Open \textit{filename} in Vim\\
\verb|vi +| \textit{filename} & Open \textit{filename} at EOF\\
\end{tabular}

\section{Leaving Vim}
\begin{tabular}{@{}ll@{}}
\verb|:x| & Save changes and exit\\
\verb|:q!| & Quit without saving\\
\end{tabular}

\section{Getting Help}
\begin{tabular}{@{}ll@{}}
\verb|:help| & Show help screen\\
\verb|:help |$x$ & Help for command $x$\\
\end{tabular}

\section{Files}
\begin{tabular}{@{}ll@{}}
\verb|:e| \textit{filename} & Edit \textit{filename} in new buffer\\
\verb|:w| \textit{filename} & Write (save as) \textit{filename}\\
\verb|:e!| & Discard unsaved changes\\
\end{tabular}

\section{Moving Around}
\begin{tabular}{@{}ll@{}}
\verb|h j k l| & $\leftarrow$ $\downarrow$ $\uparrow$ $\rightarrow$\\
\verb|g|\textit{direction} & Move one display line in \textit{direction}\\
\verb|w| & Start of next word\\
\verb|e| & End of next word\\
\verb|b| & Back a word\\
\verb|{| & Back a paragraph\\
\verb|}| & Forward a paragraph\\
\verb|1G| & Go to first line\\
\verb|G| & Go to last line\\
$n$\verb|G| or \verb|:|$n$ & Go to line $n$\\
\verb|%| & Matching parenthesis
\end{tabular}

\section{Inserting Text}
\begin{tabular}{@{}ll@{}}
\verb|i| & Insert mode \\
\verb|o| & New line below current\\
\verb|Esc| & Exit insert mode\\
\verb|u| & Undo\\
\verb|Ctrl + r| & Redo\\
\verb|.| & Repeat last command\\
\end{tabular}

\section{Moving Text}
\begin{tabular}{@{}ll@{}}
\verb|v| & Visual selection mode\\
\verb|Ctrl + v| & Block visual mode\\
\verb|I| & Insert before cursor\\
\verb|y| & Yank (copy) text\\
\verb|yy| & Yank entire line\\
\verb|x| & Kill char under cursor\\
\verb|d|$m$ & Kill text for movement $m$\\
\verb|dd| & Kill entire line\\
\verb|D| & Kill line after cursor\\
\verb|>| & Shift right\\
\verb|<| & Shift left\\
\verb|p| & Paste
\end{tabular}

\section{Searching}
\begin{tabular}{@{}ll@{}}
\verb!/!\textit{term}    & Search for \textit{term} \\
\verb!n! & Next match\\
\verb|N| & Next match backwards\\
\verb|:%s/|\textit{query}\verb|/|\textit{replace}\verb|/g| & Substitute \textit{replace} for \textit{query}\\
\verb|:%s/|\textit{query}\verb|/|\textit{replace}\verb|/gc| & With confirmation\\
\end{tabular}

\section{Buffers}
\begin{tabular}{@{}ll@{}}
\verb!:e! \textit{filename}    & Edit \textit{filename} in new buffer \\
\verb|:b| \textit{filename} & Switch to buffer \textit{filename}\\
\verb|:bn| & Next buffer\\
\verb|:bp| & Previous buffer\\
\verb|:bd| & Delete (close) buffer\\
\verb|Ctrl + wv| & Split window vertically\\
\verb|Crtl + ww| & Switch windows\\
\verb|Ctrl + wq| & Quit window\\
\end{tabular}

\section{Macros}
\begin{tabular}{@{}ll@{}}
\verb|q|\textit{letter} & Start recording macro to \textit{letter}\\
\verb|q| & Stop recording macro\\
\verb|@|\textit{letter} & Run macro \textit{letter}\\
\verb|@@| & Run macro again\\
$n$\verb|@|\textit{letter} & Run $n$ times\\
\end{tabular}
